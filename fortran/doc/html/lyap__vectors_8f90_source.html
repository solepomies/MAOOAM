66  !> Initialize Lyapunov computation (possibly also vectors in later version)
67  !> and initializes also a random orthogonal matrix for the matrix ensemble.
68  SUBROUTINE init_lyap
69  INTEGER :: AllocStat,ilaenv,info
70  lwork=ilaenv(1,"dgeqrf"," ",ndim,ndim,ndim,-1)
71  lwork=ndim*lwork
72  ALLOCATE(prop_buf(ndim,ndim),lyapunov(ndim),loclyap(ndim),ensemble(ndim,ndim),tau(ndim),prop(ndim,ndim), &
73  & work2(ndim),work(lwork),stat=allocstat)
74  IF (allocstat /= 0) stop "*** Not enough memory ! ***"
75 
76  lyapunov=0.0d0
77  loclyap=0.0d0
78  CALL init_one(prop)
79  CALL random_number(ensemble)
80  ensemble=2*(ensemble-0.5)
81  CALL dgeqrf(ndim,ndim,ensemble,ndim,tau,work,lwork, info) ! qr decomposition
82  END SUBROUTINE init_lyap

84  !> Multiplies prop_mul from the left with the prop matrix defined in this

91  END SUBROUTINE multiply_prop

93  !> Performs the benettin step in integration. Multiplies the aggregated
94  !> propagators in prop with ensemble and performs QR decomposition (Gram-Schmidt
95  !> orthogonalization gives Q and upper triangular matrix R). Computes also the
96  !> Lyapunov exponents via the diagonal of R. WATCH OUT: prop is changed during
97  !> the subroutine and restored to a unit matrix
98  SUBROUTINE benettin_step
99  INTEGER :: info,k
100 
101  ! Multiply the Propagator prop from the right side with the non transposed q matrix
102  ! from the qr decomposition which is stored in ensemble.
103  CALL dorm2r("r","n",ndim,ndim,ndim,ensemble,ndim,tau,prop,ndim,work2,info)
104  ! prop contains prop*ensemble but QR decomposed(tau is needed for that as
105  ! well !) => copy to ensemble
106  ensemble=prop
107 
108  ! From here on ensemble contains the new information prop*ensemble
109  CALL dgeqrf(ndim,ndim,ensemble,ndim,tau,work,lwork, info) ! qr decomposition
110 
111  DO k=1,ndim
112  loclyap(k)=log(abs(ensemble(k,k)))/rescaling_time
113  END DO
114 
115  !
116  ! Add here save for
117  !
118 
119  ! Initialise prop again with unit matrix
120  CALL init_one(prop)
121 
122  END SUBROUTINE benettin_step
