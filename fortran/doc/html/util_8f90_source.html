60  !> Random generator initialization routine
61  SUBROUTINE init_random_seed()
62  USE iso_fortran_env, only: int64
63  USE ifport !, only: getpid
64  IMPLICIT NONE
65  INTEGER, ALLOCATABLE :: seed_loc(:)
66  INTEGER :: i, n, un, istat, dt(8), pid
67  INTEGER(int64) :: t
68 
69  CALL random_seed(size = n)
70  ALLOCATE(seed_loc(n))
71  ! First try IF the OS provides a random number generator
72  OPEN(newunit=un, file="/dev/urandom", access="stream", &
73  form="unformatted", action="read", status="old", iostat=istat)
74  IF (istat == 0) THEN
75  READ(un) seed_loc
76  CLOSE(un)
77  ELSE
78  ! Fallback to XOR:ing the current time and pid. The PID is
79  ! useful in case one launches multiple instances of the same
80  ! program in parallel.
81  CALL system_clock(t)
82  IF (t == 0) THEN
83  CALL date_and_time(values=dt)
84  t = (dt(1) - 1970) * 365_int64 * 24 * 60 * 60 * 1000 &
85  + dt(2) * 31_int64 * 24 * 60 * 60 * 1000 &
86  + dt(3) * 24_int64 * 60 * 60 * 1000 &
87  + dt(5) * 60 * 60 * 1000 &
88  + dt(6) * 60 * 1000 + dt(7) * 1000 &
89  + dt(8)
90  END IF
91  pid = getpid()
92  t = ieor(t, int(pid, kind(t)))
93  DO i = 1, n
94  seed_loc(i) = lcg(t)
95  END DO
96  END IF
97  CALL random_seed(put=seed_loc)

99  ! This simple PRNG might not be good enough for real work, but is
100  ! sufficient for seeding a better PRNG.
101  FUNCTION lcg(s)
102  integer :: lcg
103  integer(int64) :: s
104  IF (s == 0) THEN
105  s = 104729
106  ELSE
107  s = mod(s, 4294967296_int64)
108  END IF
109  s = mod(s * 279470273_int64, 4294967291_int64)
110  lcg = int(mod(s, int(huge(0), int64)), kind(0))
111  END FUNCTION lcg
112  END SUBROUTINE init_random_seed
